// Print the symbolic ractangle

public class symbol {
    public static void main(String[] args) {
        int rows = 5;

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < rows; j++) {
                if (j % 2 == 0) {
                    System.out.print("$");
                } else {
                    System.out.print("h");
                }
            }
            System.out.println();
        }
    }
}
